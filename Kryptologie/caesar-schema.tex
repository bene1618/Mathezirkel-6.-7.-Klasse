\documentclass{zusammenfassung}
\graphicspath{ {./illustrationen/} }
\usepackage{etoolbox}
\usepackage{alphalph}
\usepackage{intcalc}
\geometry{left=1cm,right=1cm}
\usetikzlibrary{matrix}

\begin{document}

\begin{center}
  \begin{tikzpicture}
    \let\firstline\empty
    \foreach\x in {1,...,26}{%
      \edef\txt{\AlphAlph{\x}}%
      \gappto\firstline{\bf}%
      \expandafter\gappto\expandafter\firstline\expandafter{\txt\&}%
    }%
    \gappto\content{\\}%
    \let\content\empty
    \foreach\y in {0,...,25}{%
      \edef\yp{\intcalcAdd{\y}{25}}%
      \foreach\x in {\y,...,\yp}{%
        \edef\j{\AlphAlph{\intcalcInc{\intcalcMod{\x}{26}}}}%
        \expandafter\gappto\expandafter\content\expandafter{\j\&}%
      }
      \gappto\content{\\}%
    }
    \matrix [matrix of nodes, ampersand replacement=\&]
    {
      \firstline \\
      \content \\
    };
  \end{tikzpicture}
\end{center}

\end{document}
